16  package de.axnsoftware.settings;
17  
18  import java.lang.annotation.Documented;
19  import java.lang.annotation.ElementType;
20  import java.lang.annotation.Retention;
21  import java.lang.annotation.RetentionPolicy;
22  import java.lang.annotation.Target;
23  
24  /**
25   * The annotation Property must be present on all fields of a
26   * {@link PropertyClass} which should be read from or written to the backend
27   * store. The system will ignore all fields of a property class that have not
28   * been annotated by this.
29   *
30   * @author Carsten Klein "cklein" <carsten.klein@axn-software.de>
31   * @since 1.0.0
32   */
33  @Documented
34  @Retention(RetentionPolicy.RUNTIME)
35  @Target(
36          {
37      ElementType.FIELD, ElementType.TYPE
38  })
39  public @interface Property
40  {
41  
42      /**
43       * The key specifies the relative key by which the property is addressed in
44       * the underlying store.
45       *
46       * By default, the annotated field's name will be used. In order to override
47       * the default behaviour, a custom key must be specified.
48       *
49       * Please note that the specified key must adhere to the standard Java
50       * package name production rules.
51       *
52       * @return the specified key or the empty string
53       */
54      String key() default "";
55  
56      /**
57       * For simple types, e.g. Integer or String, and custom types for which
58       * there exists a type mapper, one can define a default value. The default
59       * value will be set when loading the {@link PropertyClass} via its
60       * associated {@link ISettingsStore}.
61       *
62       * @return the default value as a string or the empty string
63       */
64      String defaultValue() default "";
65  
66      /**
67       * For custom types, which have not been annotated by the
68       * {@link PropertyClass} annotation, custom type mappers {@link ITypeMapper}
69       * can be specified.
70       *
71       * @return the fully qualified class name of the type mapper implementation
72       */
73      String typeMapper() default "";
74  }
