16  package de.axnsoftware.settings.impl.accessor;
17  
18  import java.util.Map;
19  
20  /**
21   * The abstract class AbstractMapItemAccessorImpl models the root of a hierarchy
22   * of derived implementation classes and it provides the default behaviour for
23   * all implementations of the {@code IContainerItemAccessor} interface for all
24   * {@code Map} like properties.
25   *
26   * @author Carsten Klein "cklein" <carsten.klein@axn-software.de>
27   * @since 1.0.0
28   */
29  public abstract class AbstractMapItemAccessorImpl extends AbstractContainerItemAccessorImpl<String> {
30  
31      protected AbstractMapItemAccessorImpl() {
32          super();
33      }
34  
35      /**
36       * {@inheritDoc}
37       */
38      @Override
39      public Object getValue(final Object settingsRoot) {
40          Map<String, Object> container = (Map<String, Object>) this.getParentAccessor().getValue(settingsRoot);
41          return container.get(this.getItemKey());
42      }
43  
44      /**
45       * {@inheritDoc}
46       */
47      @Override
48      public void setValue(final Object value, final Object settingsRoot) {
49          Map<String, Object> container = (Map<String, Object>) this.getParentAccessor().getValue(settingsRoot);
50          container.put(this.getItemKey(), value);
51      }
52  }
