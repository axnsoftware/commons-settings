16  package de.axnsoftware.settings.integration.customtype;
17  
18  import de.axnsoftware.settings.IBackingStore;
19  import de.axnsoftware.settings.ITypeMapper;
20  import java.util.prefs.BackingStoreException;
21  
22  /**
23   *
24   * @author Carsten Klein "cklein" <carsten.klein@axn-software.de>
25   */
26  public class CustomTypeTypeMapperImpl
27          implements ITypeMapper
28  {
29  
30      @Override
31      public Object copyOf(Object value)
32      {
33          CustomType original = (CustomType) value;
34          return new CustomType(original.getIntValue(), original.getFloatValue());
35      }
36  
37      @Override
38      public Object readFromBackingStore(IBackingStore backingStore, String key,
39                                         Class<?> type) throws
40              BackingStoreException
41      {
42          return new CustomType(backingStore.getInteger(key + ".int"),
43                                backingStore.getFloat(key + ".float"));
44      }
45  
46      @Override
47      public Object valueOf(String value, Class<?> type)
48      {
49          String[] components = value.split("|");
50          return new CustomType(Integer.valueOf(components[0]), Float.valueOf(
51                  components[1]));
52      }
53  
54      @Override
55      public void writeToBackingStore(IBackingStore backingStore, String key,
56                                      Object value) throws BackingStoreException
57      {
58          CustomType original = (CustomType) value;
59          backingStore.setFloat(key + ".float", original.getFloatValue());
60          backingStore.setInteger(key + ".int", original.getIntValue());
61      }
62  }
